    <h1 id="contador"></h1>

    <script>

        const dataInicio = new Date("2024-02-25T00:00:00");

        function atualizarContador() {

            const agora = new Date();

            const diferenca = agora - dataInicio;

            const meses = Math.floor(diferenca / (1000 * 60 * 60 * 24 * 30));

            const dias = Math.floor((diferenca / (1000 * 60 * 60 * 24)) % 30);

            const horas = Math.floor((diferenca / (1000 * 60 * 60)) % 24);

            const minutos = Math.floor((diferenca / (1000 * 60)) % 60);

            const segundos = Math.floor((diferenca / 1000) % 60);

            document.getElementById("contador").innerText =

                `${meses} meses, ${dias} dias, ${horas} horas, ${minutos} minutos, ${segundos} segundos`;

        }

        setInterval(atualizarContador, 1000);

    </script>
